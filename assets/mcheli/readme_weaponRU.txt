
;***********************************************************************************
;■ Конфигурационный файл оружия: weapons/***.txt, sound/***_snd.ogg
;***********************************************************************************

; ★ Важно ★
; Конфигурационные файлы оружия можно перезагружать даже во время работы Minecraft.
; Садитесь в транспортное средство → Нажмите клавишу R, чтобы открыть экран снабжения → Опции мода → Разработка → Перезагрузить все оружия
; Это перезагрузит все оружия, включая оружие, которое держит игрок.

; Для добавления оружия необходимы следующие 2 файла (все в нижнем регистре):
; ・Добавьте конфигурационный файл оружия (txt) в папку weapons
; ・Добавьте звуковой файл оружия с тем же именем + _snd в папку sound
; Например, для оружия с именем abc, вам нужны файлы weapons/abc.txt и sound/abc_snd.ogg
; ※ Начиная с версии 0.9.4, звуковые файлы больше не являются обязательными.

; Некоторые числовые параметры имеют верхние и нижние пределы.

DisplayName = M134 Minigun
; Отображаемое название (используйте только символы и символы в полушироком формате)

Type = MachineGun1
; Тип оружия, выберите один из следующих:
; MachineGun1  : Стационарный пулемет ... M134
; MachineGun2  : Пулемет, выровненный по виду игрока ... M230
; Torpedo      : Торпеда, движущаяся к цели под водой ... Mk46
; CAS          : Ближняя авиационная поддержка для целевой области ... A-10
; Rocket       : Стационарная неуправляемая ракета ... Hydra70 или SNEB68mm
; ASMissile    : Ракета, нацеливающаяся на наземную область ... AGM119
; AAMissile    : Ракета, отслеживающая летающих мобов ... AIM92
; TVMissile    : Ракета, которую игрок может управлять после запуска ... AGM114[TV]
; ATMissile    : Ракета, отслеживающая наземных мобов ... AGM114
; Bomb         : Бомба, сбрасываемая прямо вниз ... CBU-100
; MkRocket     : Ракета, нацеленная на область удара для артиллерийского удара ... Hydra 70mm M264RP
; Dummy        : Не функциональное оружие, используется для отображения текста в слоте оружия
; Smoke        : Создает след от дыма ... Дым белый
; Dispenser    : Использует предмет в месте удара ... Водяной диспенсер (Till)
; TargetingPod : Отмечает мобов, игроков и блоки ... targeting_pod_block

Power = 8
; Урон

DamageFactor = tank, 2.0
; Множитель урона
; Первый параметр может быть установлен как один из следующих:
; player   : Игрок
; heli или helicopter : Вертолет
; plane    : Самолет с фиксированным крылом
; tank     : Танки или автомобили
; vehicle  : Наземные транспортные средства
; Второй параметр — множитель урона [0~]. Например, если Power равно 10, а множитель 3.4, то урон будет равен 34.
; Эта настройка применяется, если написано несколько строк.

Acceleration = 4.0
; Скорость пули (максимум 4.0 для большинства, но MachineGun1, MachineGun2 и Rocket могут достигать до 100.0)

AccelerationInWater = 4.0
; Скорость торпед под водой (максимум 4.0)

VelocityInWater = 0.5
; Ускорение под водой
; Под водой это значение умножается на скорость каждый тик.

Explosion = 0
; Мощность взрыва при попадании (0 = без взрыва, 1 = мощность газовой оболочки, 2~)

ExplosionInWater = 0
; Мощность взрыва при попадании под водой

ExplosionBlock = 0
; Мощность разрушения блоков при попадании. 0 означает отсутствие разрушения блоков

ExplosionAltitude = 10
; Высота от земли, на которой произойдет взрыв
; Установка на 10 означает, что взрыв произойдет, если он находится в пределах 10 м от земли

DelayFuse = 30
; Задержка сгорания: отсчитывает время до исчезновения пули после попадания
; Если Explosion/ExplosionInWater не равны 0, то произойдет взрыв при исчезновении

Bound = 0.4
; Сила отскока при попадании
; Если используется Bound, DelayFuse также должен быть настроен, в противном случае взрыв произойдет сразу после попадания.

TimeFuse = 30
; Время сгорания: отсчитывает время до исчезновения пули после запуска
; Если Explosion/ExplosionInWater не равны 0, то произойдет взрыв при исчезновении

Flaming = false
; Распространять ли огонь при попадании (false = выключено, true = включено)
; ※ Включено только при Explosion > 0

Sight = MoveSight или None или MissileSight
; Тип прицела, отображаемого на экране
; MoveSight  : Прицел, который двигается в соответствии с ориентацией транспортного средства
; MissileSight : Прицел с возможностью захвата цели для мобов (необходим для AAMissile/ATMissile)

Zoom = 4.2, 9.2
; Только для переносных оружий
; Уровень увеличения при отсутствии прицела. Если значений несколько, их можно переключать с помощью клавиши Z.

Group = MainGun
; Установите группу оружия
; Когда одно оружие в одной группе используется, все они будут перезагружены.
; Например, если вы разделите главный пистолет танка на три типа снарядов, как указано ниже,
; rehinmetall_apfsds.txt, rehinmetall_he.txt, canistershell.txt, и установите Group = MainGun,
; то при использовании одного, все остальные тоже будут перезагружены. Боеприпасы не уменьшатся.
; 1-й: rehinmetall_apfsds
; 2-й: rehinmetall_he
; 3-й: canistershell
; Эта настройка предотвращает немедленную смену оружия после использования первого.

Delay=5
; Время ожидания перед следующим использованием (в единицах примерно 1/20 секунды), меньше значит быстрее.

ReloadTime=80
; Время ожидания до завершения перезарядки (в единицах 1/20 секунды), меньше значит быстрее.
; *При установке времени перезарядки в 0, установите количество боеприпасов на ненулевое значение.

Round=100
; Количество боеприпасов. Установите в 0, если не используется, или полностью исключите этот параметр.

SoundVolume=3
; Громкость при использовании оружия.
; В Minecraft, если установить выше 1.0, будет достигнут максимальный уровень громкости.
; (Для уменьшения громкости установите ниже 1.0)
; Если выше 1.0, звук будет слышен на большом расстоянии.

SoundPitch=1.0
; Тональность звука (от 0.0 до 1.0).

SoundPitchRandom=0.1
; Диапазон случайного изменения тональности (от 0.0 до 1.0)
; Например:
; SoundPitch = 0.8
; SoundPitchRandom = 0.2
; Результирующая тональность звука будет в пределах от 0.6 до 0.8.

SoundDelay=1
; Время ожидания между звуками при повторном выстреле.
; Без этого звуки, как у M134, могут накладываться друг на друга, и другие звуки могут быть заглушены.

Sound=rocket_snd
; Укажите файл звука. Расширение файла не требуется.
; Если это не указано, будет использован стандартный звук для оружия с именем оружия_snd.ogg.

LockTime=20
; Время блокировки для типов ракет. Большое значение означает большее время блокировки.

RidableOnly=true
; Установите значение true для блокировки только при нахождении игрока на транспортном средстве.

ProximityFuseDist=1.0
; Рабочий радиус датчика приближения для ракет с блокировкой по цели.
; При установке в 1, он взорвется на расстоянии 1 метра.

RigidityTime=0
; Время отсчета перед началом отслеживания ракеты после выстрела.
; Если не указано, используется значение по умолчанию 7.

Accuracy=1
; Точность для неуправляемых пуль или ракет. Чем выше значение, тем меньше ошибка.

Bomblet=25
; Количество суббоеприпасов, которое будет развернуто после использования. Используется для кассетных бомб.

BombletSTime=5
; Время до развертывания суббоеприпасов.

BombletDiff=0.7
; Степень разброса суббоеприпасов.

ModeNum=2
; Количество режимов оружия, которые можно переключать с помощью клавиши X.
; Применимо только для некоторых типов оружия:
; Type = MachineGun2 → Переключение на бронебойные снаряды (взрывчатые снаряды). Отключено, если Explosion = 0.
; Type = TVMissile → Переключение на обычную управляемую ракету (не-TV ракета).
; Type = ATMissile → Переключение на режим TA (Top Attack).
; Type = Rocket → Переключение на снаряды HEIAP (сброс нескольких суббоеприпасов в воздухе).
; Разрешено только 1 или 2 режима.
; По умолчанию используется 1, если не указано другое.

Piercing=2
; Количество блоков, которые может пробить оружие.

HeatCount=20
; Количество тепла, которое генерируется при использовании оружия с нагревом ствола.

MaxHeatCount=150
; Максимальная теплот capacity.

FAE=true
; Включение/выключение взрывчатых веществ с топливом (FAE).
; Установите значение true, чтобы включить FAE.
; FAE не уничтожает блоки.

ModelBullet=bullet
ModelBomblet=cbc
; Укажите файл модели для пуль и суббоеприпасов.
; В этом случае используются models/bullets/bullet.obj и textures/bullets/bullet.png для пуль.
; Суббоеприпасы кассетной бомбы будут использовать models/bullets/cbc.obj и textures/bullets/cbc.png.

Destruct = true
; Включение самоуничтожения транспортного средства при использовании
; Доступно только при Type = Bomb.
; Эффективно только если транспортное средство является БПЛА.

Gravity = -0.04
GravityInWater = 0.0
; Скорость падения боевой части.
; Чем больше абсолютное значение, тем быстрее она падает.
; GravityInWater — это скорость падения в воде.

GuidedTorpedo = true
; Переключение между управляемыми/неуправляемыми торпедами.
; При установке в true, торпеда будет нацеливаться на указанный блок.
; При установке в false, торпеда будет неуправляемой и будет двигаться прямо от точки удара.

TrajectoryParticle = flame
; Указывает эффект "следа", когда используется определенное оружие (например, частицы, когда ракета отслеживает цель).
; Опции включают:
; none          ... Без эффекта
; explode       ... Эффект дыма
; flame         ... Пламя
; hugeexplosion ... Огромный взрыв
; largeexplode  ... Большой взрыв
; largesmoke    ... Большой дым
; smoke         ... Дым
;
; Для опытных пользователей: Это строка, используемая в spawnParticle, поэтому можно указать другие строки.
; Частицы устарели с версии 1.0.0, используйте AddMuzzleFlash или AddMuzzleFlashSmoke вместо этого.

TrajectoryParticleStartTick = 10
; Количество тикетов перед началом эффекта TrajectoryParticle.

DisableSmoke = true
; Отключает дымовые эффекты для определенных оружий
; (Не для эффекта выстрела, а для дыма при движении).

AddMuzzleFlash = 0.5, 0.20, 1, 150,254,219,184
; AddMuzzleFlash = Расстояние от начала, Размер, Время отображения, A, R, G, B
; ★ Примечание: Если интервал использования оружия около 5, отображение может не работать правильно, поэтому избегайте этого.

AddMuzzleFlashSmoke = 2.2, 1, 5.0, 2.0, 15, 180,250,245,240
; AddMuzzleFlashSmoke = Расстояние от начала, Количество отображений, Размер, Диапазон, Время отображения, A, R, G, B
; ★ Примечание: Если интервал использования оружия около 5, отображение может не работать правильно, поэтому избегайте этого.

SetCartridge = cartridge, 0.0, 0, 0, 2.00, -0.04, 0.40
; Настройки для выпадения пустых патронов при использовании оружия
; SetCartridge = название_модели, Ускорение, Поворот по горизонтали, Поворот по вертикали, МасштабМодели, Гравитация, Отскок
; название_модели : Название модели (строчные буквы, полудоступное)
; Ускорение : Сила броска (0 означает падение прямо вниз)
; Поворот по горизонтали : Угол в горизонтальном направлении от оружия (положительное: влево, отрицательное: вправо)
; Поворот по вертикали : Угол в вертикальном направлении от оружия (положительное: вниз, отрицательное: вверх)
; МасштабМодели : Масштаб отображения
; Гравитация : Сила гравитации
; Отскок : Отскок при попадании в блок

MaxAmmo = 40

; Для достижения максимума в 40 патронов будет использовано следующее:
; 12 железных слитков, 16 пороха, 8 редстоунов.

BulletColor = 255, 255, 255, 255

SmokeColor = 230, 200, 20, 80

SmokeMaxAge = 500
; Максимальная продолжительность отображения дыма.

FixCameraPitch = true

CameraRotationSpeedPitch = 0.3
; Множитель скорости вращения камеры (меньшие значения позволяют точнее настраивать точку попадания).

DispenseItem = flint_and_steel

; Увеличение множителя отдачи во время отдачи ускоряет время отдачи.

Target = monsters/others
